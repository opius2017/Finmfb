  export FRONTEND_POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app={{ .Release.Name }}-frontend,release={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")

  kubectl port-forward $FRONTEND_POD_NAME 8081:80 --namespace {{ .Release.Namespace }}

To check the status of your deployment:

For detailed logs:
  kubectl logs -f deployment/{{ .Release.Name }}-api -n {{ .Release.Namespace }}
  kubectl logs -f deployment/{{ .Release.Name }}-frontend -n {{ .Release.Namespace }}

For scaling:
  kubectl scale deployment {{ .Release.Name }}-api --replicas=5 -n {{ .Release.Namespace }}
  kubectl scale deployment {{ .Release.Name }}-frontend --replicas=5 -n {{ .Release.Namespace }}
